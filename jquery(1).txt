1. Dynamic To-Do List with jQuery
Problem:
Create a To-Do List where users can:
•	Add new tasks using an input field.
•	Delete tasks by clicking a "Remove" button.
•	Toggle task completion (strike-through) when clicked.
Requirements:
•	Use jQuery event delegation for dynamically added tasks.
•	Apply fadeIn() when adding and fadeOut() when removing tasks.
ANSWER:
<!DOCTYPE html>
<html>
<head>
    <title>Dynamic To-Do List</title>
    <script src="https://code.jquery.com/jquery-3.6.0.min.js"></script>
    <style>
        body {
            font-family: Arial;
            text-align: center;
        }

        #taskInput {
            padding: 8px;
            width: 200px;
        }

        #addBtn {
            padding: 8px 12px;
        }

        ul {
            list-style: none;
            padding: 0;
        }

        li {
            margin: 10px auto;
            padding: 10px;
            width: 300px;
            background: #f4f4f4;
            display: none; /* for fadeIn */
        }

        .completed {
            text-decoration: line-through;
            color: gray;
        }

        button.remove {
            float: right;
            background: red;
            color: white;
            border: none;
            padding: 5px;
            cursor: pointer;
        }
    </style>
</head>

<body>

    <h2>To-Do List</h2>

    <input type="text" id="taskInput" placeholder="Enter task">
    <button id="addBtn">Add</button>

    <ul id="taskList"></ul>

    <script>
        $(document).ready(function(){

            // Add Task
            $("#addBtn").click(function(){
                let task = $("#taskInput").val().trim();

                if(task !== ""){
                    let newTask = `
                        <li>
                            <span class="taskText">${task}</span>
                            <button class="remove">Remove</button>
                        </li>
                    `;

                    $("#taskList").append(newTask);
                    $("#taskList li:last").fadeIn(); // fadeIn effect
                    $("#taskInput").val("");
                }
            });

            // Toggle completion (Event Delegation)
            $("#taskList").on("click", ".taskText", function(){
                $(this).toggleClass("completed");
            });

            // Remove Task (Event Delegation)
            $("#taskList").on("click", ".remove", function(){
                $(this).parent().fadeOut(function(){
                    $(this).remove();
                });
            });

        });
    </script>

</body>
</html>

        }

<h2>Image Slider</h2>

<div id="slider">
    <div class="slide"><img src="https://picsum.photos/id/1015/500/300"></div>
    <div class="slide"><img src="https://picsum.photos/id/1016/500/300"></div>
    <div class="slide"><img src="https://picsum.photos/id/1018/500/300"></div>
    <div class="slide"><img src="https://picsum.photos/id/1020/500/300"></div>
</div>

<button id="prev">Previous</button>
<button id="next">Next</button>

<script>
$(document).ready(function(){

    let slides = $(".slide");
    let index = 0;
    let interval;

    // Show first image
    slides.eq(index).show();

    function showSlide(i){
        slides.fadeOut();
        slides.eq(i).fadeIn();
    }

    function nextSlide(){
        index = (index + 1) % slides.length;
        showSlide(index);
    }

    function prevSlide(){
        index = (index - 1 + slides.length) % slides.length;
        showSlide(index);
    }

    // Auto-play
    function startAuto(){
        interval = setInterval(nextSlide, 3000);
    }

    function stopAuto(){
        clearInterval(interval);
    }

    startAuto();

    // Buttons
    $("#next").click(function(){
        nextSlide();
    });

    $("#prev").click(function(){
        prevSlide();
    });

    // Pause on hover
    $("#slider").hover(stopAuto, startAuto);

});
</script>

</body>
</html>

<body>

<p id="score">Score: 0</p>

<div id="quiz">

    <!-- Question 1 -->
    <div class="question" data-answer="Mobile">
        <h3>Which is a portable device?</h3>
        <div class="options">
            <button>Mobile</button>
            <button>Table</button>

        </div>
    </div>

    <!-- Question 2 -->
    <div class="question" data-answer="Laptop">
        <h3>Which is used for computing?</h3>
        <div class="options">
            <button>Fan</button>
            <button>Laptop</button>
            <button>Pen</button>
        </div>
    </div>

    <!-- Question 3 -->
    <div class="question" data-answer="Camera">
        <h3>Which is used to capture photos?</h3>
        <div class="options">
            <button>Camera</button>
            <button>Book</button>
            <button>Bottle</button>
        </div>
    </div>

</div>

<script>
$(document).ready(function(){

    let index = 0;
    let score = 0;
    let time = 10;
    let timer;

    let questions = $(".question");

    function showQuestion(i){
        time = 10;
        $("#timer").text("Time: " + time);

        questions.hide();
        $(questions[i]).fadeIn();

        startTimer();
    }

    function startTimer(){
        clearInterval(timer);

        timer = setInterval(function(){
            time--;
            $("#timer").text("Time: " + time);

            if(time === 0){
                nextQuestion();
            }
        }, 1000);
    }

    function nextQuestion(){
        clearInterval(timer);

        $(questions[index]).delay(200).fadeOut(function(){
            index++;

            if(index < questions.length){
                showQuestion(index);
            } else {
                $("#quiz").hide();
                $("#timer").hide();
                $("#result").html("Final Score: " + score).fadeIn();
            }
        });
    }

    $(".options button").click(function(){
        let selected = $(this).text();
        let correct = $(questions[index]).data("answer");

        if(selected === correct){
            score++;
            $("#score").text("Score: " + score);
        }

        nextQuestion();
    });

    // Start first question
    showQuestion(index);

});
</script>

</body>
</html>
