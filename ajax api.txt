// 🔑 Replace with your API key
const API_KEY = "d9fca50f7e312bf64523ea3b57db7082";

function getWeather() {

  const city = document.getElementById("city").value.trim();
  const result = document.getElementById("result");

  if (!city) {
    result.innerHTML = '<span class="err">Please enter a city name</span>';
    return;
  }

  result.innerHTML = "Loading...";

  const xhr = new XMLHttpRequest();

  const url = "https://api.openweathermap.org/data/2.5/weather?q=" +
              encodeURIComponent(city) +
              "&appid=" + API_KEY +
              "&units=metric";

  xhr.open("GET", url, true);

  xhr.onload = function () {

    if (xhr.status === 200) {
      const data = JSON.parse(xhr.responseText);

      result.innerHTML = `
        <h3>${data.name}, ${data.sys.country}</h3>
        <p><b>Temperature:</b> ${data.main.temp} °C</p>
        <p><b>Condition:</b> ${data.weather[0].main}</p>
        <p><b>Humidity:</b> ${data.main.humidity}%</p>
        <p><b>Wind Speed:</b> ${data.wind.speed} m/s</p>
      `;

    } else if (xhr.status === 404) {
      result.innerHTML = '<span class="err">City not found</span>';

    } else if (xhr.status === 401) {
      result.innerHTML = '<span class="err">Invalid or inactive API key</span>';

    } else {
      result.innerHTML = '<span class="err">Error fetching data</span>';
    }
  };

  xhr.onerror = function () {
    result.innerHTML = '<span class="err">Network error</span>';
  };

  xhr.send();
}

// Press Enter support

  if (e.key === "Enter") getWeather();

<script>
const search = document.getElementById('search');
const drop   = document.getElementById('dropdown');
const load   = document.getElementById('loading');
let timer;
search.addEventListener('input', function () {
  clearTimeout(timer);
  const q = search.value.trim().toLowerCase();
  if (q === '') {
    drop.style.display = 'none';
    drop.innerHTML = '';
    return;
  }
  // Debounce so we only fire after 250ms idle
  timer = setTimeout(function () { fetchUsers(q); }, 250);
});
function fetchUsers(query) {
  load.style.display = 'block';
  drop.style.display = 'none';
  const xhr = new XMLHttpRequest();
  xhr.open('GET', 'https://jsonplaceholder.typicode.com/users', true);
  xhr.onload = function () {
    load.style.display = 'none';
    if (xhr.status === 200) {
      const all = JSON.parse(xhr.responseText);
      const matches = all.filter(function (u) {
        return u.name.toLowerCase().indexOf(query) !== -1 ||
               u.username.toLowerCase().indexOf(query) !== -1;

      });
      renderResults(matches);
    } else {
      drop.innerHTML = '<div class="err">Error ' + xhr.status + '. Please try again.</div>';
      drop.style.display = 'block';
    }
  };
  xhr.onerror = function () {
    load.style.display = 'none';
    drop.innerHTML = '<div class="err">Network error. Check your connection.</div>';
    drop.style.display = 'block';
  };
  xhr.send();
}
function renderResults(matches) {
  if (matches.length === 0) {
    drop.innerHTML = '<div class="item"><em>No users found</em></div>';
  } else {
    drop.innerHTML = matches.map(function (u) {
      return '<div class="item"><strong>' + u.name + '</strong>' +
             '<br><small>@' + u.username + ' &middot; ' + u.email + '</small></div>';
    }).join('');
  }
  drop.style.display = 'block';
}
</script>
</body>
</html>

  <div class="input-row">
    <input type="text" id="newTask" placeholder="Add a new task...">

  </div>
  <div id="spinner"><span class="spin"></span> Loading...</div>
  <div id="taskList"></div>
<script>
const API = 'https://jsonplaceholder.typicode.com/todos';
const list    = document.getElementById('taskList');
const spinner = document.getElementById('spinner');
function showSpinner(on) { spinner.style.display = on ? 'block' : 'none'; }
function loadTasks() {
  showSpinner(true);
  fetch(API + '?_limit=5')
    .then(function (r) { return r.json(); })
    .then(function (data) {
      list.innerHTML = '';
      data.forEach(renderTask);
    })
    .catch(function () { list.innerHTML = '<p style="color:red">Failed to load tasks.</p>'; })
    .finally(function () { showSpinner(false); });
}
function renderTask(task) {
  const div = document.createElement('div');
  div.className = 'task';
  div.id = 'task-' + task.id;

  div.innerHTML = '<span>' + task.title + '</span>' +
                  '<button class="del" onclick="deleteTask(' + task.id + ')">Delete</button>';
  list.appendChild(div);
}
function addTask() {
  const title = document.getElementById('newTask').value.trim();
  if (title === '') return;
  showSpinner(true);
  fetch(API, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ title: title, completed: false, userId: 1 })
  })
    .then(function (r) { return r.json(); })
    .then(function (newTask) {
      // JSONPlaceholder returns id 201 for any POST -- that's fine for the demo.
      // Make IDs unique within the session by using a timestamp.
      newTask.id = Date.now();
      renderTask(newTask);
      document.getElementById('newTask').value = '';
    })
    .catch(function () { alert('Failed to add task.'); })
    .finally(function () { showSpinner(false); });
}
function deleteTask(id) {
  showSpinner(true);
  fetch(API + '/' + id, { method: 'DELETE' })
    .then(function () {
      const el = document.getElementById('task-' + id);
      if (el) el.remove();
    })
    .catch(function () { alert('Failed to delete task.'); })
    .finally(function () { showSpinner(false); });
}
loadTasks();
</script>
</body>
</html>

accepts POST/DELETE requests and returns realistic responses, but data is not actually persisted on the server. The
UI manages local state to simulate persistence within the session.

// Mock data - in a real app this would be on the server, fetched via AJAX
const DATA = {
  IN: {
    'Maharashtra':  ['Mumbai', 'Pune', 'Nagpur'],
    'Karnataka':    ['Bangalore', 'Mysore'],
    'Tamil Nadu':   ['Chennai', 'Coimbatore']
  },
  US: {
    'California':   ['Los Angeles', 'San Francisco', 'San Diego'],
    'Texas':        ['Houston', 'Austin', 'Dallas'],
    'New York':     ['New York City', 'Buffalo']
  },
  UK: {

// Simulate an AJAX call. In production:
//   fetch('/api/states?country=' + code).then(r => r.json()).then(callback)
function ajaxCall(payload, callback, errorCallback) {
  setTimeout(function () {
    if (payload) callback(payload);
    else errorCallback();
  }, 300);
}
function loadStates() {
  const code = document.getElementById('country').value;
  const stateSel = document.getElementById('state');
  const citySel  = document.getElementById('city');
  const status   = document.getElementById('stateStatus');
  // Reset dependent dropdowns
  stateSel.innerHTML = '<option value="">-- Loading... --</option>';
  stateSel.disabled = true;
  citySel.innerHTML = '<option value="">-- Select City --</option>';
  citySel.disabled = true;
  document.getElementById('cityStatus').textContent = '';
  if (code === '') {
    stateSel.innerHTML = '<option value="">-- Select State --</option>';
    status.textContent = '';
    return;
  }
  ajaxCall(DATA[code],
    function (states) {
      const keys = Object.keys(states);
      if (keys.length === 0) {
        stateSel.innerHTML = '<option value="">-- No states found --</option>';
        status.textContent = 'No states available for this country.';
        status.className = 'status err';
        return;
      }
      stateSel.innerHTML = '<option value="">-- Select State --</option>';
      keys.forEach(function (s) {
        stateSel.innerHTML += '<option value="' + s + '">' + s + '</option>';
      });
      stateSel.disabled = false;
      status.textContent = keys.length + ' state(s) loaded.';
      status.className = 'status';
    },
    function () {
      stateSel.innerHTML = '<option value="">-- Error --</option>';
      status.textContent = 'Failed to load states.';
      status.className = 'status err';
    }
  );
}
function loadCities() {
  const country = document.getElementById('country').value;
  const state   = document.getElementById('state').value;
  const citySel = document.getElementById('city');
  const status  = document.getElementById('cityStatus');
  citySel.innerHTML = '<option value="">-- Loading... --</option>';
  citySel.disabled = true;
  if (state === '') {
    citySel.innerHTML = '<option value="">-- Select City --</option>';
    status.textContent = '';
    return;
  }
  ajaxCall(DATA[country] && DATA[country][state],
    function (cities) {

      if (!cities || cities.length === 0) {
        citySel.innerHTML = '<option value="">-- No cities found --</option>';
        status.textContent = 'No cities available.';
        status.className = 'status err';
        return;
      }
      citySel.innerHTML = '<option value="">-- Select City --</option>';
      cities.forEach(function (c) {
        citySel.innerHTML += '<option value="' + c + '">' + c + '</option>';
      });
      citySel.disabled = false;
      status.textContent = cities.length + ' cities loaded.';
      status.className = 'status';
    },
    function () {
      citySel.innerHTML = '<option value="">-- Error --</option>';
      status.textContent = 'Failed to load cities.';
      status.className = 'status err';
    }
  );
}

<script>
const API  = 'https://jsonplaceholder.typicode.com/comments';
const list = document.getElementById('commentList');
function loadComments() {
  fetch(API + '?_limit=5')
    .then(function (r) { return r.json(); })
    .then(function (data) {
      list.innerHTML = '';
      data.forEach(function (c) { addCommentToDOM(c.name, c.body); });

function addCommentToDOM(name, body) {
  const div = document.createElement('div');
  div.className = 'comment';
  div.innerHTML = '<div class="name">' + escapeHtml(name) + '</div>' +
                  '<div>' + escapeHtml(body) + '</div>';

    requestAnimationFrame(function () { div.classList.add('show'); });

function escapeHtml(s) {

      addCommentToDOM(name, body);
      document.getElementById('name').value = '';
      document.getElementById('body').value = '';
    })
    .catch(function () { err.textContent = 'Failed to post comment.'; });
}

$(document).ready(function () {
  var total = 0;
  var nextItemId = 1;
  // Fetch product data from mock API. In a real shop you'd hit /api/products
  // and get real product objects. JSONPlaceholder /photos gives us titles+ids.
  $.get('https://jsonplaceholder.typicode.com/photos?_limit=6', function (data) {
    $('#productList').empty();
    data.forEach(function (p) {
      // Make up a price based on id so it's deterministic
      var price = parseFloat((10 + (p.id % 50)).toFixed(2));
      var $item = $('<div class="product"></div>')
        .append('<span><strong>' + p.title.substring(0, 30) + '</strong>'
                + '<br><small>$' + price + '</small></span>')
        .append($('<button>Add to Cart</button>')
          .click(function () { addToCart(p.title.substring(0, 30), price); }));
      $('#productList').append($item);
    });
  }).fail(function () {

    $('#productList').html('<p style="color:red">Failed to load products.</p>');
  });
  function addToCart(name, price) {
    // Simulate POST to /api/cart in a real app
    $.ajax({
      url: 'https://jsonplaceholder.typicode.com/posts',
      method: 'POST',
      data: JSON.stringify({ title: name, price: price }),
      contentType: 'application/json',
      success: function () {
        var id = 'cart-' + (nextItemId++);
        var $item = $('<div class="cart-item" id="' + id + '"></div>')
          .append('<span>' + name + ' &mdash; $' + price.toFixed(2) + '</span>')
          .append($('<button class="remove">Remove</button>')
            .click(function () { removeFromCart(id, price); }));
        $item.hide().appendTo('#cartList').fadeIn(300);
        total += price;
        updateTotal();
      },
      error: function () { alert('Failed to add item.'); }
    });
  }
  function removeFromCart(id, price) {
    $('#' + id).fadeOut(300, function () {
      $(this).remove();
      total -= price;
      updateTotal();
    });
  }
  function updateTotal() {
    $('#total').text('Total: $' + total.toFixed(2));
  }
});
</script>

<script>
$(document).ready(function () {
  var page = 1;
  var perPage = 5;
  var loading = false;
  var done = false;
  function loadPosts() {
    if (loading || done) return;
    loading = true;
    $('#loader').show();
    $.ajax({
      url: 'https://jsonplaceholder.typicode.com/posts',
      data: { _page: page, _limit: perPage },
      success: function (data) {
        $('#loader').hide();
        loading = false;
        if (data.length === 0) {
          done = true;
          $('#end').show();
          return;
        }
        data.forEach(function (p) {
          var $div = $('<div class="post"></div>')
            .append('<h3>' + p.title + '</h3>')
            .append('<p>' + p.body + '</p>');

          $('#posts').append($div);
          $div.fadeIn(400);
        });
        page++;
        // JSONPlaceholder has 100 posts; stop after we've loaded them all
        if (page * perPage > 100) { done = true; $('#end').show(); }
      },
      error: function () {
        $('#loader').hide();
        loading = false;
        $('#posts').append('<p style="color:red;text-align:center;">Failed to load posts.</p>');
      }
    });
  }
  // Trigger load when user scrolls near the bottom
  $(window).scroll(function () {
    if ($(window).scrollTop() + $(window).height() >= $(document).height() - 100) {
      loadPosts();
    }
  });
  loadPosts();  // Initial load
});
</script>
</body>
</html>

// ✅ MOCK API DATA (acts like fetched data)
const API_DATA = {
  question: "Which frontend framework do you like?",
  options: [
    { id: 1, text: "React", votes: 5 },
    { id: 2, text: "Angular", votes: 3 },
    { id: 3, text: "Vue", votes: 4 }
  ]
};

// Simulate AJAX fetch
function fetchPoll() {
  return new Promise(resolve => {
    setTimeout(() => resolve(API_DATA), 500);
  });
}

// Load poll
async function loadPoll() {
  const data = await fetchPoll();

  document.getElementById("question").innerText = data.question;

  const container = document.getElementById("options");
  container.innerHTML = "";

  data.options.forEach(opt => {
    const div = document.createElement("div");
    div.className = "poll-option";
    div.innerHTML = `
      ${opt.text}
      <div class="bar" id="bar-${opt.id}"></div>
      <span id="count-${opt.id}">${opt.votes} votes</span>
    `;

    div.onclick = () => vote(opt.id);
    container.appendChild(div);
  });

  updateBars();
}

// Vote function (AJAX simulation)
function vote(id) {

  // ❌ Prevent multiple votes
  if (localStorage.getItem("voted")) {
    document.getElementById("msg").innerText = "You already voted!";
    return;
  }

  document.getElementById("msg").innerText = "";

  // Simulate AJAX POST
  setTimeout(() => {
    const option = API_DATA.options.find(o => o.id === id);
    option.votes++;

    localStorage.setItem("voted", "true");

    animateUpdate();
  }, 300);
}

// Update UI with animation
function animateUpdate() {
  const container = document.getElementById("options");

  // slide up
  container.style.transform = "translateY(-10px)";
  container.style.opacity = "0";

  setTimeout(() => {
    loadPoll();

    // slide down
    container.style.transform = "translateY(0)";
    container.style.opacity = "1";
  }, 300);
}

// Update bar widths
function updateBars() {
  const total = API_DATA.options.reduce((sum, o) => sum + o.votes, 0);

  API_DATA.options.forEach(opt => {
    const percent = (opt.votes / total) * 100;

    document.getElementById(`bar-${opt.id}`).style.width = percent + "%";
    document.getElementById(`count-${opt.id}`).innerText = opt.votes + " votes";
  });
}

// Reset for demo
function resetVote() {
  localStorage.removeItem("voted");
  document.getElementById("msg").innerText = "Vote reset!";
}

// Initial load
loadPoll();

</script>
